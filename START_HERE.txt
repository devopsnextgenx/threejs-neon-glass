📊 DATA FORMAT
═══════════════════════════════════════════════════════════════════════════

{
  "nodes": [{
    "id": "unique-id",           ← Unique identifier
    "label": "Display Name",     ← Shown in visualization
    "community": 0,              ← Color group (0-3)
    "weight": 0.8,               ← Size multiplier (0-1)
    "source_file": "path/file",  ← Metadata
    "source_location": "L42"     ← Metadata
  }],
  "links": [{
    "source": "node-id",         ← Source node ID
    "target": "node-id",         ← Target node ID
    "relation": "calls"          ← Relationship type
  }]
}
